<script>
  const inputs = document.querySelectorAll("input");

  inputs.forEach(input => input.addEventListener("input", calculate));

  function calculate() {
    const clients = +document.getElementById("clients").value;
    const adoption = +document.getElementById("adoption").value / 100;
    const servers = +document.getElementById("servers").value;
    const price = +document.getElementById("price").value;
    const storage = +document.getElementById("storage").value;
    const costPerGB = +document.getElementById("cost").value;

    const activeClients = clients * adoption;
    const totalServers = activeClients * servers;

    const mrr = totalServers * price;
    const arr = mrr * 12;

    const totalStorage = totalServers * storage;
    const monthlyCost = totalStorage * costPerGB;

    const profit = mrr - monthlyCost;
    const margin = mrr ? (profit / mrr) * 100 : 0;

    document.getElementById("mrr").innerText = `$${mrr.toFixed(0)}`;
    document.getElementById("arr").innerText = `$${arr.toFixed(0)}`;
    document.getElementById("costTotal").innerText = `$${monthlyCost.toFixed(0)}`;
    document.getElementById("profit").innerText = `$${profit.toFixed(0)}`;
    document.getElementById("margin").innerText = `${margin.toFixed(1)}%`;
  }

  calculate();
</script>
